hostname=zedboard

kernel_addr_r=0x3000000
expand_bootfile=setenv bootfile ${hostname}/uImage

fdt_addr_r=0x2a00000
expand_fdtfile=setenv fdtfile ${hostname}/zedboard-zynq7.dtb

# this needs to be loaded from mmc and into fpga before the kernel,
# .since I use the same address with the kernel
# .try also: 0x100000
#setenv bitstream_addr_r=0x3000000
bitstream_addr_r=0x100000
expand_bitstream=setenv bitstream ${hostname}/fpga.bin

ipaddr=192.168.42.12
serverip=192.168.42.1
gatewayip=192.168.42.1
netmask=255.255.255.0

# --> uImage from tftp
kernel_netload=tftpboot ${kernel_addr_r} ${bootfile}
# <-- uImage from tftp

# --> fdt from tftp
fdt_netload=tftpboot ${fdt_addr_r} ${fdtfile}
# <-- fdt from tftp

# --> bitstream from tftp and into fpga
bitstream_netload_fpgaload=if tftpboot ${bitstream_addr_r} ${bitstream}; then fpga load 0 ${bitstream_addr_r} ${filesize}; else echo no ${bitstream}; fi
# <-- bitstream from tftp and into fpga

# --> uImage and fdt from tftp
netload=run bitstream_netload_fpgaload kernel_netload fdt_netload
# <-- uImage and fdt from tftp

# --> generic bootargs
# default:
bootargs=console=ttyPS0,115200n8
# bootgraph:
#bootargs=initcall_debug printk.time=y console=ttyPS0,115200n8
# bootchart:
#bootargs=initcall_debug printk.time=y init=/sbin/bootchartd console=ttyPS0,115200n8
# <-- generic bootargs

# --> rootfs from nfs
nfsroot=/opt/poky/zedboard-rootfs
ips_to_bootargs=setenv bootargs ${bootargs} ip=${ipaddr}:${serverip}:${gatewayip}:${netmask}:${hostname}:${netdev}:off
nfs_to_bootargs=setenv bootargs ${bootargs} nfsroot=${serverip}:${nfsroot},v3,tcp
default_to_bootargs=setenv bootargs ${bootargs} noinitrd nohlt panic=1
compose_nfs_bootargs=run ips_to_bootargs; run nfs_to_bootargs; run default_to_bootargs
nfsboot=echo Booting from nfs...; run compose_nfs_bootargs

compose_nfsboot=run nfsboot
# <-- rootfs from nfs

compose_default=run expand_bitstream; echo bitstream: ${bitstream}; run expand_bootfile; echo bootfile: ${bootfile}; run expand_fdtfile; echo fdtfile: ${fdtfile}; ping ${serverip}
uenvcmd=run compose_default; run netload; run compose_nfsboot; printe bootargs; bootm ${kernel_addr_r} - ${fdt_addr_r}
